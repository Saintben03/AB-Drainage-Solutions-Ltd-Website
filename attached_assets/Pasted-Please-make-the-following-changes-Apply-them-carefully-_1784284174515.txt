## CHANGE 3: Generate real brand favicons from the existing logo assets

Run this in the Replit shell from the project root directory:

```bash
cd artifacts/ab-drainage && npm install sharp --save-dev
```

Then run this Node.js script in the shell (copy and paste the whole thing):

```bash
node --input-type=module << 'EOF'
import sharp from 'sharp';
import { createRequire } from 'module';
import path from 'path';
import fs from 'fs';
import { fileURLToPath } from 'url';

const root = process.cwd();

const logos = [
  { name: 'favicon-construction.png', file: 'AB-Construction-300x54_1783288669800.png' },
  { name: 'favicon-fencing.png',      file: 'AB-Fencing-300x56_1783288689329.png' },
  { name: 'favicon-facilities.png',   file: 'AB-Facilities-scaled-1-300x52_1783288632041.png' },
];

function findFile(filename, searchDir) {
  const entries = fs.readdirSync(searchDir, { withFileTypes: true });
  for (const e of entries) {
    const full = path.join(searchDir, e.name);
    if (e.isDirectory()) {
      const found = findFile(filename, full);
      if (found) return found;
    } else if (e.name === filename) {
      return full;
    }
  }
  return null;
}

const outDir = path.join(root, 'public');
const srcDir = path.join(root, 'src');

for (const { name, file } of logos) {
  const src = findFile(file, srcDir);
  if (!src) { console.error('NOT FOUND:', file); continue; }

  const meta = await sharp(src).metadata();
  const size = meta.height;

  const out = path.join(outDir, name);
  await sharp(src)
    .extract({ left: 0, top: 0, width: size, height: size })
    .resize(64, 64)
    .png()
    .toFile(out);

  console.log('Created:', out, `(cropped ${size}x${size} → 64x64)`);
}
EOF
```

Make sure you run that from INSIDE `artifacts/ab-drainage/` directory.
