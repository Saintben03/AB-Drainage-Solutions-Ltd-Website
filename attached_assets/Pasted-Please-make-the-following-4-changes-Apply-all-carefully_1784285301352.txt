## CHANGE 3: Fix favicons for division domains

### Step A — Run in the Replit shell from inside artifacts/ab-drainage/:

```bash
npm install sharp --save-dev
```

Then paste this entire block into the shell:

```bash
node --input-type=module << 'EOF'
import sharp from 'sharp';
import path from 'path';
import fs from 'fs';

const root = process.cwd();

function findFile(name, dir) {
  for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
    const full = path.join(dir, entry.name);
    if (entry.isDirectory()) { const r = findFile(name, full); if (r) return r; }
    else if (entry.name === name) return full;
  }
  return null;
}

const logos = [
  { out: 'favicon-construction.png', file: 'AB-Construction-300x54_1783288669800.png' },
  { out: 'favicon-fencing.png',      file: 'AB-Fencing-300x56_1783288689329.png' },
  { out: 'favicon-facilities.png',   file: 'AB-Facilities-scaled-1-300x52_1783288632041.png' },
];

for (const { out, file } of logos) {
  const src = findFile(file, path.join(root, 'src'));
  if (!src) { console.error('NOT FOUND:', file); continue; }
  const meta = await sharp(src).metadata();
  const size = meta.height;
  const outPath = path.join(root, 'public', out);
  await sharp(src).extract({ left: 0, top: 0, width: size, height: size }).resize(64, 64).png().toFile(outPath);
  console.log('✓ Created', out);
}
EOF
```

Confirm it prints 3 lines starting with ✓ before continuing.
